# 构建图床应用

# 1. 创建vue工程

vue create img-web

# 2. 创建node（express）工程

express img-server

# 3. 创建数据库

  1. 按照字段值，创建数据库（须事先创建table_id_seq）
-- ----------------------------
-- Table structure for pictures
-- ----------------------------
DROP TABLE IF EXISTS "public"."pictures";
CREATE TABLE "public"."pictures" (
  "id" int2 NOT NULL DEFAULT nextval('table_id_seq'::regclass),
  "name" varchar(255) COLLATE "pg_catalog"."default",
  "url" varchar(255) COLLATE "pg_catalog"."default",
  "createdAt" date,
  "updatedAt" date,
  "deletedAt" date
)
;

-- ----------------------------
-- Primary Key structure for table pictures
-- ----------------------------
ALTER TABLE "public"."pictures" ADD CONSTRAINT "pictures_pkey" PRIMARY KEY ("id");
  1. 设置id的自增
-- 创建自增序列
CREATE SEQUENCE table_id_seq START WITH 1 INCREMENT BY 1 NO MINVALUE NO MAXVALUE CACHE 1;

-- 修改id列（删除记录不影响排序）
ALTER TABLE pictures ALTER COLUMN ID 
SET DEFAULT nextval( 'table_id_seq' );
