    # Anaconda

    基本信息

    # 产看版本
conda --version

# 查看安装的包
conda list

# 安装包 移除包 更新包
conda install package_name==1.12.0
conda remove package_name
conda update package_name

    创建环境相关

    # 查看所有环境
conda env list

# 创建环境
# 其中 -n 代表 name，env_name 是需要创建的环境名称，list of packages 则是列出在新环境中需要安装的工具包。
# 示例 conda create --name myenv python 安装最新版本python（包含基本依赖包certifi pip setuptools wheel wincertstore）
# 示例 conda create -n py2 python=2.7 pandas 安装2.7版本的python（包含基本依赖包，pandas相关依赖包）
conda create -n env_name list_of_packages

#测试
conda create -n Pointnet_Pointnet2_pytorch python=3.7
conda install pytorch torchvision torchaudio cudatoolkit=11.3 -c pytorch


# 删除环境
conda env remove -n env_name

# 激活虚拟换环境
activate env_name

# 退出虚拟环境
deactivate

    导出依赖项

    # 导出运行依赖（进入某环境后）
conda env export > environment.yaml

# 使用上述yaml文件，创建新环境（自动创建一个与之前名字一样的虚拟环境）
conda env create -f environment.yaml

# 使用上述yaml文件，创建新环境（指定新环境的名字）
conda env create -f environment.yml -n new_env_name
