    # Git

    # 切换远程仓库地址

    # 移除现在的地址
git remote rm origin 
# 添加新的地址
git remote add origin https://gitee.com/gisleung/gisleung_notes.git

    # 添加GitHub本地SSH验证

    # 如果公钥出问题，需要删除C:\Users\gisleung\.ssh下文件id_rsa和id_rsa.pub，再重新生成ssh公钥即可解决，

# 1.1 如果同时删除了known_hosts，需要先重设置用户名和邮箱
$ git config --global user.name "gisleung"
$ git config --global user.email "gisleung@163.com"

# 1.2 如果未删除，上面两步可以跳过。查看配置，
$ git config --list
$ git config user.name
$ git config user.email

# 2 重新生成ssh公钥。一路yes和回车，自动生成id_rsa和id_rsa.pub，用记事本打开id_rsa.pub 将全部的内容复制 
$ ssh-keygen -t rsa -C "gisleung@163.com"

# 3 网页打开https://github.com/，进入个人setting中新增SSH keys，粘贴进去

# 4 验证
$ ssh -T git@github.com
